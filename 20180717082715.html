DROP TABLE IF EXISTS country;
CREATE TABLE country (
id int(11) NOT NULL AUTO_INCREMENT,
countryname varchar(255) DEFAULT NULL,
countrycode varchar(255) DEFAULT NULL,
PRIMARY KEY (id)
